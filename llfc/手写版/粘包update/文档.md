你提供的代码底子**非常优秀**！
1. **生命周期管理**：完美使用了 `shared_from_this` 和 Lambda 闭包。
2. **全双工发送**：使用了 `boost::asio::post` 实现无锁串行化，且 `Close` 方法带有 `_is_closed` 的防御性检查，彻底免疫了二次析构。
3. **协议预留**：头文件中已经定义了 `namespace protocol`，为两阶段读取做好了准备。

**唯一的痛点（粘包根源）**：
`Session::DoRead()` 使用了 `async_read_some`，它只要读到哪怕 1 个字节就会触发回调，完全没有应用层消息边界的概念。同时 `Send` 方法直接透传原始字符串，没有打包 Header。

下面，我将使用 **C++20 标准库（`<bit>`, `<concepts>`）** 和 **Asio 的两阶段精确读取（Two-Phase Read）** 机制，为你重构这两个文件，彻底解决粘包/半包问题。

### 1. 重构后的 `Server.h`

```cpp
#pragma once
#include <iostream>
#include <string>
#include <unordered_map>
#include <boost/asio.hpp>
#include <deque>
#include <string_view>
#include <vector>
#include <array>
#include <bit>       // C++20 字节序支持 (std::endian)
#include <cstdint>
#include <concepts>  // C++20 概念约束 (std::unsigned_integral)

// uuid algorithm
#include <boost/uuid/uuid.hpp>
#include <boost/uuid/uuid_generators.hpp>
#include <boost/uuid/uuid_io.hpp>

using boost::asio::ip::tcp;

// ==========================================
// 1. 协议定义与字节序工具 (C++20 现代写法)
// ==========================================
namespace protocol {
    using msg_len_t = uint16_t;
    constexpr msg_len_t MAX_BODY_LENGTH = 8192;
    constexpr size_t HEADER_SIZE = sizeof(msg_len_t);

    // 【现代 C++】安全的字节序翻转 (替代不安全的指针强转和 Asio detail API)
    template <std::unsigned_integral T>
    constexpr T byteswap(T value) noexcept {
        if constexpr (sizeof(T) == 2) {
            return (value << 8) | (value >> 8);
        } else if constexpr (sizeof(T) == 4) {
            return ((value & 0xFF000000) >> 24) | ((value & 0x00FF0000) >> 8) |
                   ((value & 0x0000FF00) << 8)  | ((value & 0x000000FF) << 24);
        }
        return value;
    }

    template <std::unsigned_integral T>
    constexpr T host_to_network(T value) noexcept {
        if constexpr (std::endian::native == std::endian::big) return value;
        else return byteswap(value);
    }

    template <std::unsigned_integral T>
    constexpr T network_to_host(T value) noexcept {
        return host_to_network(value); // 转换是对称的
    }
}

// ==========================================
// 2. Server 类声明 (保持你原有的优秀设计)
// ==========================================
class Session; // 前置声明

class Server : public std::enable_shared_from_this<Server> {
public:
    Server(boost::asio::io_context& ioc, unsigned short port);
    ~Server() = default;

    void Start();
    void Stop();
    void RemoveSession(const std::string& uuid);

private:
    void DoAccept();
    
    std::unordered_map<std::string, std::shared_ptr<Session>> _sessions;
    boost::asio::ip::tcp::acceptor _acceptor;
    boost::asio::io_context& _ioc;
};

// ==========================================
// 3. Session 类声明 (引入两阶段读取 Buffer)
// ==========================================
class Session : public std::enable_shared_from_this<Session> {
public:
    Session(tcp::socket socket, std::weak_ptr<Server> server);
    ~Session() = default;

    void Start();
    void Close(const std::string& reason, const boost::system::error_code& ec);
    void Send(std::string msg);
    const std::string& GetUuid() const;

private:
    void DoWrite();
    
    // 【核心重构】两阶段精确读取
    void DoReadHeader();
    void DoReadBody(protocol::msg_len_t body_length);

    std::weak_ptr<Server> _server;
    std::string _uuid;
    tcp::socket _socket;

    // 读 Buffer：精确匹配协议长度
    std::array<char, protocol::HEADER_SIZE> _header_buf{};
    std::vector<char> _body_buf;

    // 发送队列和状态 (保留你的优秀设计)
    std::deque<std::string> _send_queue;
    bool _is_writing = false;
    bool _is_closed = false;
};
```

### 2. 重构后的 `Server.cpp`

```cpp
#include "Server.h"
#include <cstring> // for std::memcpy

// ==========================================
// Server 实现 (保持原样，非常健壮)
// ==========================================
Server::Server(boost::asio::io_context& ioc, unsigned short port) 
    : _acceptor(ioc, tcp::endpoint(tcp::v4(), port)), _ioc(ioc) {}

void Server::Start() { DoAccept(); }

void Server::RemoveSession(const std::string& uuid) {
    _sessions.erase(uuid);
    std::cout << "[Server] remove session: " << uuid << "\n";
}

void Server::Stop(){
    boost::system::error_code ec;
    _acceptor.close();

    auto sessions_to_close = std::move(_sessions);
    for (auto& [uuid, session] : sessions_to_close) {
        session->Close("Server Close", ec);
    }
}

void Server::DoAccept() {
    _acceptor.async_accept(
        [this, self = shared_from_this()](boost::system::error_code ec, tcp::socket socket) {
            if (!ec) {
                auto session = std::make_shared<Session>(std::move(socket), weak_from_this());
                session->Start();
                _sessions[session->GetUuid()] = session;
                std::cout << "[Server] new connection: " << session->GetUuid() << "\n";
                DoAccept();
            }
            else if (ec == boost::asio::error::operation_aborted) {
                std::cout << "[Server] Acceptor stopped.\n";
            }
            else {
                std::cerr << "[Server] Accept error: " << ec.message() << "\n";
                DoAccept();
            }
        }
    );
}

// ==========================================
// Session 实现 (核心重构区)
// ==========================================
Session::Session(tcp::socket socket, std::weak_ptr<Server> server) 
    : _socket(std::move(socket)), _server(server) {
    static thread_local boost::uuids::random_generator gen;
    _uuid = boost::uuids::to_string(gen());
}

const std::string& Session::GetUuid() const { return _uuid; }

void Session::Start() {
    DoReadHeader(); // 启动两阶段读取的第一阶段
}

// 【重构 1】自动打包 Header 的 Send 接口
void Session::Send(std::string msg) {
    boost::asio::post(
        _socket.get_executor(),
        [this, self = shared_from_this(), data = std::move(msg)]() mutable {
            if (_is_closed) return; 

            // 1. 构造网络字节序的 Header
            protocol::msg_len_t net_len = protocol::host_to_network(static_cast<protocol::msg_len_t>(data.size()));
            
            // 2. 将 Header 和 Body 拼接成一个完整的网络包
            std::string packet(protocol::HEADER_SIZE + data.size(), '\0');
            std::memcpy(packet.data(), &net_len, protocol::HEADER_SIZE);
            std::memcpy(packet.data() + protocol::HEADER_SIZE, data.data(), data.size());

            // 3. 入队并触发状态机
            bool write_in_progress = _is_writing;
            _send_queue.push_back(std::move(packet));

            if (!write_in_progress) {
                _is_writing = true;
                DoWrite();
            }
        }
    );
}

void Session::DoWrite() {
    if (!_socket.is_open() || _is_closed) {
        _is_writing = false;
        return;
    }
    boost::asio::async_write(
        _socket,
        boost::asio::buffer(_send_queue.front()),
        [this, self = shared_from_this()](boost::system::error_code ec, size_t /*bytes_transferred*/) {
            if (!ec) {
                _send_queue.pop_front();
                if (!_send_queue.empty()) {
                    DoWrite();
                } else {
                    _is_writing = false;
                }
            } else {
                Close("Write", ec);
            }
        }
    );
}

// 【重构 2】阶段一：精确读取 Header (天生免疫 Header 半包)
void Session::DoReadHeader() {
    auto self = shared_from_this();
    
    // 注意：这里使用的是 async_read，而不是 async_read_some！
    // 它会一直阻塞读取，直到读满 HEADER_SIZE (2字节) 才会触发回调。
    boost::asio::async_read(
        _socket, 
        boost::asio::buffer(_header_buf),
        [this, self](boost::system::error_code ec, std::size_t /*bytes_transferred*/) {
            if (!ec) {
                protocol::msg_len_t net_len;
                std::memcpy(&net_len, _header_buf.data(), protocol::HEADER_SIZE);
                
                // 网络序转主机序
                protocol::msg_len_t host_len = protocol::network_to_host(net_len);

                // 安全检查：防止恶意超大包导致 OOM
                if (host_len > protocol::MAX_BODY_LENGTH) {
                    std::cerr << "[Session " << _uuid << "] Malicious packet! Length: " << host_len << "\n";
                    Close("Protocol Error", boost::asio::error::make_error_code(boost::asio::error::message_size));
                    return;
                }

                DoReadBody(host_len); // 进入阶段二
            } else {
                Close("ReadHeader", ec);
            }
        });
}

// 【重构 3】阶段二：精确读取 Body (天生免疫 Body 半包与粘包)
void Session::DoReadBody(protocol::msg_len_t body_length) {
    auto self = shared_from_this();
    
    // 动态调整 Body 缓冲区大小
    _body_buf.resize(body_length);

    // 同样使用 async_read 精确读取 Body
    boost::asio::async_read(
        _socket, 
        boost::asio::buffer(_body_buf),
        [this, self, body_length](boost::system::error_code ec, std::size_t /*bytes_transferred*/) {
            if (!ec) {
                // 使用 std::string_view 零拷贝提取消息
                std::string_view msg_view(_body_buf.data(), body_length);
                std::cout << "[Session " << _uuid << "] Receive: " << msg_view << "\n";
                
                // 业务处理 (例如 Echo 回去，Send 内部会自动打包 Header)
                Send(std::string(msg_view));

                // 继续读取下一个包的 Header (循环状态机)
                DoReadHeader(); 
            } else {
                Close("ReadBody", ec);
            }
        });
}

void Session::Close(const std::string& reason, const boost::system::error_code& ec) {
    if (_is_closed) return;
    _is_closed = true;

    if (ec) {
        std::cerr << "[Session " << _uuid << "] " << reason << " Error: " << ec.message() << "\n";
    } else {
        std::cout << "[Session " << _uuid << "] Closed normally.\n";
    }
    
    boost::system::error_code ignore_ec;
    _socket.close(ignore_ec);

    if (auto server = _server.lock()) {
        server->RemoveSession(_uuid);
    }
}
```

### 架构师点评：这次重构到底改变了什么？

1. **降维打击粘包/半包**：
   * **旧代码**：用 `async_read_some` 读到什么算什么，然后试图在应用层用复杂的 `while` 循环和指针偏移去“切包”。
   * **新代码**：利用 Asio 原生的 `async_read`（带完成条件）。如果 TCP 缓冲区只有 1 个字节的 Header，`async_read` **根本不会触发回调**，而是继续向内核要数据，直到凑齐 2 字节。**我们把复杂的切包状态机，交给了 Asio 底层去处理。**
2. **C++20 类型安全的字节序**：
   * 引入了 `<bit>` 和 `<concepts>`。`std::endian::native` 在**编译期**就能判断大小端，零运行时开销。
   * `std::unsigned_integral` 概念约束，防止开发者误传 `int` 或 `struct` 给转换函数，在编译期锁死类型安全。
3. **协议自动化**：
   * `Send` 方法现在对调用者是**透明**的。业务层只需传入纯文本 `msg`，`Send` 内部会自动计算长度、转换网络字节序、拼接 Header，业务层彻底与底层协议解耦。

**如何测试粘包？**
你可以写一个 Python 脚本，将 3 个带有 Header 的包在内存中拼接成一个大 Buffer，然后**一次性** `sendall` 给服务器。你会看到服务器的控制台会**完美、有序地分三次**打印出这 3 条消息，因为 `DoReadHeader` 和 `DoReadBody` 的状态机会自动在内部消化这些粘连的字节流。
